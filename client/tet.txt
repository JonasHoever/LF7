#include <SPI.h>
#include <MFRC522.h>

#define SS_PIN 10
#define RST_PIN 9

MFRC522 mfrc522(SS_PIN, RST_PIN);

void setup() {
  Serial.begin(9600);
  SPI.begin();
  mfrc522.PCD_Init();
  mfrc522.PCD_SetAntennaGain(mfrc522.RxGain_max);
  
  // Warte bis Serial bereit ist
  while (!Serial) {
    delay(10);
  }
  
  delay(1000); // Kurze Pause für Stabilität
}

void loop() {
  // Suche nach neuen NFC-Tags
  if (mfrc522.PICC_IsNewCardPresent() && mfrc522.PICC_ReadCardSerial()) {
    
    // Baue die Tag-ID als HEX-String zusammen
    String tagID = "";
    for (byte i = 0; i < mfrc522.uid.size; i++) {
      if (mfrc522.uid.uidByte[i] < 0x10) {
        tagID += "0";
      }
      tagID += String(mfrc522.uid.uidByte[i], HEX);
    }
    
    // In Großbuchstaben umwandeln
    tagID.toUpperCase();
    
    // Sende NUR die Tag-ID an Python
    Serial.println(tagID);
    
    // Warte 2 Sekunden um Mehrfach-Scans zu verhindern
    delay(2000);
    
    // Warte bis Tag weggenommen wird
    while (mfrc522.PICC_IsNewCardPresent()) {
      delay(100);
    }
  }
  
  // Kurze Pause im Loop
  delay(100);
}
